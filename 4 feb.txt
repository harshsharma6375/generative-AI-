write a c program input character :
		a :
			two number input :addition 

		s:
			two number input :substraction

		d :
			two number input :divide

		m :
			two number input :multiply

		otherwise :input is not matched 

#include<stdio.h>
int main(){
    char a ;
    printf("enter the character");
    scanf ("%c",&a);
    if (a == 'a' )
    {
        int a, b , c;
        printf("addition of a and b :");
        scanf("%d %d",&a,&b);
        c = a+b;
        printf("%d",c);
        
    }
    else if (a == 's')
    {
        int d , e , f ;
        printf ("subtractioin of d and e :");
        scanf("%d %d",&d,&e);
        f = d - e;
        printf("%d",f);
        
        
    }
    else if(a == 'm')
    {
        int g,h,i;
        printf("multiplication of g and h :");
        scanf("%d %d",&g ,&h);
        i = g * h;
        printf("%d",i);
        
    }
    else if  (a == 'd')
    {
        int j,k,l;
        
        printf("division of j and k :");
        scanf("%d %d",&j ,&k);
        l = j / k ;
        printf("%d",l);
        
        
    }
    else {
        printf("not valid");
    }
    return 0 ;
    
}
